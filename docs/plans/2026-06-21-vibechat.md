# VibeChat 实现计划（最终版）

> **For agentic workers:** 用 superpowers:subagent-driven-development 或 executing-plans 逐任务实现。步骤用 `- [ ]` 复选框跟踪。

**Goal:** 一天内交付 AI 驱动的情绪社交 Web 应用：用户输入 → 对话分析出「型」+经历 → 跨 session 的 memory 混合检索匹配 → 匿名 WebSocket 聊天。

**核心心智模型（来自 claw0 S05+S06）：**
- **每个用户 = 一个 session**（S05 简化骨架，砍掉 5 层路由，退化为 `user_id → session`）。session 持有该用户的 memory、状态机、room_id、聊天记录。
- **每个 session 装一条 UserMemory**（S06）= 型 + 经历结构 + keywords + VAD。
- **匹配 = 跨 session 的 hybrid_search**（S06 引擎：关键词TF-IDF + 哈希向量 + 加权合并 + 等待加权 + MMR），在 `status=waiting` 的 session 间互相检索，排除自己/已匹配。
- **ExperienceStore 沉淀经历** → 匹配同时搜「在线等待池 + 最近沉淀的经历」，解决冷启动 & 声音墙。
- **「型」是被看见的核心**：型名 + 镜子句 + 侦探句 + 同型声音，一页说清"你是谁"。
- **SOUL.md** 只定义 AI 的性格口吻（温暖、不诊断），不参与匹配。

```python
import time
from ..identity import gen_nickname, gen_color

class Session:
    def __init__(self, user_id: str, memory: dict):
        self.user_id = user_id
        self.memory = memory                      # UserMemory dict（型+经历+keywords+vad）
        self.status = "waiting"                    # analyzing→waiting→matched→chatting→ended
        self.room_id = None
        self.partner_id = None
        self.partner_nick = ""
        self.sync_points = []
        self.created_at = time.time()
        self.nickname = gen_nickname(memory.get("primary_emotion","calm"))
        self.color = gen_color()
        self.history = []

class SessionManager:
    def __init__(self):
        self.sessions: dict[str, Session] = {}
    def upsert(self, user_id: str, memory: dict) -> Session:
        s = Session(user_id, memory)
        self.sessions[user_id] = s
        return s
    def get(self, user_id: str) -> Session | None:
        return self.sessions.get(user_id)
    def waiting(self, exclude: str) -> list[Session]:
        return [s for s in self.sessions.values()
                if s.status == "waiting" and s.user_id != exclude and not s.room_id]

sessions = SessionManager()
```

## Task 6: ExperienceStore 混合检索（S06 引擎）

**设计：** 照搬 claw0 S06 的 hybrid_search——关键词TF-IDF + 哈希向量 + 加权合并 + 等待加权（替代时间衰减）+ MMR。在候选 memory 列表上检索，返回 (best, score, sync_points)。

**Files:** `matching/store.py`, Test `tests/test_store.py`

- [ ] **Step 1: store.py**

```python
import math, time

EMOTION_CN = {"anxiety":"焦虑","loneliness":"孤独","anger":"愤怒","sadness":"低落",
  "excitement":"兴奋","calm":"平静","confusion":"迷茫","emptiness":"空虚",
  "joy":"开心","frustration":"烦躁"}

def _tokens(m: dict) -> list[str]:
    return list(m.get("keywords", []) or [])

def _hash_vec(tokens: list[str], dim: int = 64) -> list[float]:
    v = [0.0]*dim
    for t in tokens:
        h = hash(t)
        for i in range(dim):
            v[i] += 1.0 if (h >> (i % 62)) & 1 else -1.0
    n = math.sqrt(sum(x*x for x in v)) or 1.0
    return [x/n for x in v]

def _vcos(a, b):
    dot = sum(x*y for x,y in zip(a,b))
    na = math.sqrt(sum(x*x for x in a)); nb = math.sqrt(sum(x*x for x in b))
    return dot/(na*nb) if na and nb else 0.0

def _kw_cos(a: list[str], b: list[str], df: dict, n: int) -> float:
    def tfidf(toks):
        tf = {}
        for t in toks: tf[t] = tf.get(t,0)+1
        return {t: c*(math.log((n+1)/(df.get(t,0)+1))+1) for t,c in tf.items()}
    x, y = tfidf(a), tfidf(b)
    common = set(x)&set(y)
    if not common: return 0.0
    dot = sum(x[k]*y[k] for k in common)
    nx = math.sqrt(sum(v*v for v in x.values())); ny = math.sqrt(sum(v*v for v in y.values()))
    return dot/(nx*ny) if nx and ny else 0.0

def _build_df(mems: list[dict]):
    df = {}
    for m in mems:
        for t in set(_tokens(m)): df[t] = df.get(t,0)+1
    return df, len(mems)

def _sync_points(me: dict, other: dict) -> list[str]:
    r = []
    if me.get("primary_emotion") == other.get("primary_emotion"):
        r.append(f"都在经历{EMOTION_CN.get(me['primary_emotion'],'')}")
    shared = [t for t in _tokens(me) if t in set(_tokens(other))][:2]
    if shared: r.append("都提到了" + "、".join(shared))
    mc = me.get("经历",{}).get("内在冲突"); oc = other.get("经历",{}).get("内在冲突")
    if mc and oc: r.append(f"都卡在相似的拉扯里")
    if 1 - abs(me.get("intensity",0)-other.get("intensity",0)) > 0.85:
        r.append("情绪强度接近")
    return r[:4]

def hybrid_match(me: dict, candidates: list[dict], created_ats: list[float]):
    """me: 我的 memory；candidates: 其他人的 memory 列表（与 created_ats 同序）。
    返回 (best_index 或 None, score, sync_points)。"""
    if not candidates:
        return None, 0.0, []
    df, n = _build_df(candidates + [me])
    me_vec = _hash_vec(_tokens(me))
    now = time.time()
    best_i, best_score = None, -1.0
    for i, c in enumerate(candidates):
        kw = _kw_cos(_tokens(me), _tokens(c), df, n)              # 关键词通道
        vec = _vcos(me_vec, _hash_vec(_tokens(c)))                # 向量通道
        merged = 0.4*vec + 0.6*kw
        # 情绪基调过滤：正负向相反则压分
        if me.get("valence",0)*c.get("valence",0) < -0.2:
            merged *= 0.5
        # 等待加权（替代时间衰减）：等越久越优先
        waited = now - created_ats[i]
        merged *= (1 + min(waited, 60)/120)                       # 最多 +50%
        if merged > best_score:
            best_i, best_score = i, merged
    return best_i, best_score, _sync_points(me, candidates[best_i])
```

- [ ] **Step 1: match.py**

```python
import uuid
from fastapi import APIRouter
from pydantic import BaseModel
from ..session.manager import sessions

router = APIRouter()
THRESH_FULL, THRESH_LOW = 0.45, 0.20   # hybrid_match 分数尺度

class MatchReq(BaseModel):
    user_id: str
    memory: dict

@router.post("/api/match")
async def match(req: MatchReq):
    if req.memory.get("safety_level") == "crisis":
        return {"status": "crisis"}
    from ..matching.store import hybrid_match
    me = sessions.upsert(req.user_id, req.memory)
    cands = sessions.waiting(exclude=req.user_id)
    if cands:
        idx, score, pts = hybrid_match(me.memory,
            [s.memory for s in cands], [s.created_at for s in cands])
        # 等待越久阈值越低
        import time
        waited = time.time() - me.created_at
        thr = THRESH_FULL if waited < 15 else THRESH_LOW
        if idx is not None and score >= thr:
            other = cands[idx]
            room = "room_" + uuid.uuid4().hex[:8]
            for a, b in ((me, other), (other, me)):
                a.room_id = room; a.status = "matched"
                a.partner_id = b.user_id; a.partner_nick = b.nickname
                a.sync_points = pts
            return {"status":"matched","room_id":room,
                    "me":{"nickname":me.nickname,"color":me.color},
                    "partner":{"nickname":other.nickname,"color":other.color},
                    "sync_points":pts}
    return {"status":"waiting","nickname":me.nickname,"color":me.color}

@router.get("/api/match/check")
async def check(user_id: str):
    s = sessions.get(user_id)
    if s and s.room_id:
        return {"status":"matched","room_id":s.room_id,
                "me":{"nickname":s.nickname,"color":s.color},
                "partner_nick":s.partner_nick,"sync_points":s.sync_points}
    return {"status":"waiting"}
```

```python
from fastapi import WebSocket

class ConnectionManager:
    def __init__(self):
        self.rooms: dict[str, dict[str, WebSocket]] = {}
    async def connect(self, room, uid, ws: WebSocket):
        await ws.accept(); self.rooms.setdefault(room, {})[uid] = ws
    def disconnect(self, room, uid):
        if room in self.rooms: self.rooms[room].pop(uid, None)
    async def broadcast(self, room, msg: dict, exclude=None):
        for uid, ws in list(self.rooms.get(room, {}).items()):
            if uid == exclude: continue
            try: await ws.send_json(msg)
            except Exception: self.disconnect(room, uid)
    def online(self, room): return len(self.rooms.get(room, {}))

manager = ConnectionManager()
```

```python
import time
from fastapi import APIRouter, WebSocket, WebSocketDisconnect
from ..chat.manager import manager
from ..chat.icebreaker import get_icebreaker
from ..session.manager import sessions

router = APIRouter()

@router.websocket("/ws/rooms/{room_id}")
async def chat_ws(ws: WebSocket, room_id: str, user_id: str, nickname: str = "匿名"):
    await manager.connect(room_id, user_id, ws)
    s = sessions.get(user_id)
    if s: s.status = "chatting"
    await manager.broadcast(room_id, {"type":"system",
        "data":{"content":f"{nickname} 加入了","online":manager.online(room_id)}})
    if manager.online(room_id) == 2:
        pts = s.sync_points if s else []
        await manager.broadcast(room_id, {"type":"icebreaker","data":{"content":get_icebreaker(pts)}})
    try:
        while True:
            msg = await ws.receive_json()
            await manager.broadcast(room_id, {"type":"chat",
                "data":{"from":user_id,"nickname":nickname,"content":msg.get("content",""),"ts":time.time()}})
    except WebSocketDisconnect:
        manager.disconnect(room_id, user_id)
        await manager.broadcast(room_id, {"type":"system",
            "data":{"content":f"{nickname} 的频率暂时离开了","online":manager.online(room_id)}})
```
